195 void Layer::CalcActDeriv(const unsigned long batchFrom, const unsigned long batchTo)
196 {
197  verify(batchFrom >= 0 && batchTo < batchSize && batchFrom <= batchTo);
198  verify(engine != NULL);
199 
200 #ifdef FRACTAL_VERBOSE
201  printf("Layer::CalcActDeriv: %s (%ld, %ld)\n",
202  GetName().c_str(), batchFrom, batchTo);
203 #endif /* FRACTAL_VERBOSE */
204 
205  Matrix<FLOAT> srcErrSub(srcErr, batchFrom, batchTo);
206  Matrix<FLOAT> actSub(act, batchFrom, batchTo);
207 
208  switch(actType)
209  {
210  case ACT_SIGMOID:
211  engine->FuncSigmoidDeriv(actSub, srcErrSub, *stream);
212  break;
213 
214  case ACT_TANH:
215  engine->FuncTanhDeriv(actSub, srcErrSub, *stream);
216  break;
217 
218  case ACT_SOFTPLUS:
219  engine->FuncSoftplusDeriv(actSub, srcErrSub, *stream);
220  break;
221 
222  case ACT_RECTLINEAR:
223  engine->FuncRectLinearDeriv(actSub, srcErrSub, *stream);
224  break;
225 
226  case ACT_LINEAR:
227  break;
228 
229  case ACT_ONE_MINUS_LINEAR:
230  engine->MatSet(srcErrSub, (FLOAT) -1, *stream);
231  break;
232 
233  case ACT_INVERSE:
234  engine->MatSet(srcErrSub, (FLOAT) -1, *stream);
235  break;
236 
237  case ACT_SOFTMAX:
238  /* Not supported */
239  verify(false);
240  break;
241 
242  default:
243  verify(false);
244  }

302 void Layer::Activation(const unsigned long batchFrom, const unsigned long batchTo)
303 {
304  verify(engine != NULL);
305 
306  Matrix<FLOAT> stateSub(state, batchFrom, batchTo);
307  Matrix<FLOAT> actSub(act, batchFrom, batchTo);
308 
309  switch(actType)
310  {
311  case ACT_BIAS:
312  engine->MatSet(actSub, (FLOAT) 1, *stream);
313  break;
314 
315  case ACT_SIGMOID:
316  engine->FuncSigmoid(stateSub, actSub, *stream);
317  break;
318 
319  case ACT_TANH:
320  engine->FuncTanh(stateSub, actSub, *stream);
321  break;
322 
323  case ACT_SOFTPLUS:
324  engine->FuncSoftplus(stateSub, actSub, *stream);
325  break;
326 
327  case ACT_RECTLINEAR:
328  engine->FuncRectLinear(stateSub, actSub, *stream);
329  break;
330 
331  case ACT_LINEAR:
332  //engine->MatCopy(stateSub, actSub, *stream);
333  act.Link(state);
334  break;
335 
336  case ACT_ONE_MINUS_LINEAR:
337  engine->MatSet(actSub, (FLOAT) 1, *stream);
338  engine->MatAdd(stateSub, actSub, (FLOAT) -1, *stream);
339  break;
340 
341  case ACT_INVERSE:
342  engine->MatSet(actSub, (FLOAT) 0, *stream);
343  engine->MatAdd(stateSub, actSub, (FLOAT) -1, *stream);
344  break;
345 
346  case ACT_SOFTMAX:
347  engine->FuncSoftmax(stateSub, actSub, *stream);
348  break;
349 
350  default:
351  verify(false);
352  }

518 void Layer::DistributeErr(Connection *conn, const unsigned long batchFrom, const unsigned long batchTo)
519 {
520  verify(engine != NULL);
521 
522  switch(stateType)
523  {
524  case AGG_SUM:
525  conn->dstErr.Link(srcErr);
526  break;
527 
528  case AGG_MULT:
529  {
530  ConnList::const_iterator iter, iter_end;
531 
532  Matrix<FLOAT> srcErrSub(srcErr, batchFrom, batchTo);
533  Matrix<FLOAT> connErrSub(conn->dstErr, batchFrom, batchTo);
534 
535  bool isFirst = true;
536 
537  iter_end = srcList.end();
538  for(iter = srcList.begin(); iter != iter_end; ++iter)
539  {
540  if(*iter == conn) continue;
541 
542  Matrix<FLOAT> srcActSub((*iter)->dstAct, batchFrom, batchTo);
543 
544  if(isFirst == true)
545  {
546  engine->MatElemMult(srcErrSub, srcActSub, connErrSub, *conn->stream);
547  isFirst = false;
548  }
549  else
550  {
551  engine->MatElemMult(connErrSub, srcActSub, connErrSub, *conn->stream);
552  }
553  }
554 
555  if(isFirst == true)
556  {
557  conn->dstErr.Link(srcErr);
558  }
559  }
560 
561  break;
562 
563  default:
564  verify(false);
565  }
