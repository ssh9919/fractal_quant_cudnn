101 #ifdef FRACTAL_USE_OMP
102  #pragma omp parallel for private(tCur, oCur, tMaxIdx, oMaxIdx, tMax, oMax, idx, err) \
103  reduction(+:sePartialSum, cePartialSum, nPartialError)
104 #endif
105  for(i = 0; i < nFrame; i++)
106  {
107  tMaxIdx = 0;
108  oMaxIdx = 0;
109  tMax = t[i * dim];
110  oMax = o[i * dim];
111 
112  for(j = 0; j < dim; j++)
113  {
114  idx = i * dim + j;
115 
116  tCur = t[idx];
117  oCur = o[idx];
118 
119  err = oCur - tCur;
120  sePartialSum += err * err;
121 
122  if(tCur > (FLOAT) 0)
123  cePartialSum -= tCur * std::log(oCur + (double) 1e-300);
124 
125  if(oCur > oMax)
126  {
127  oMax = oCur;
128  oMaxIdx = j;
129  }
130 
131  if(tCur > tMax)
132  {
133  tMax = tCur;
134  tMaxIdx = j;
135  }
136  }
137 
138  if(tMaxIdx != oMaxIdx) nPartialError++;
139  }
140 
141 
142  seSum[outputIdx] += sePartialSum;
143  ceSum[outputIdx] += cePartialSum;
144  nError[outputIdx] += nPartialError;
145 }
