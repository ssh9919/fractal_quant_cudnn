124 void Engine::MemAlloc(Mem *mem, unsigned long loc)
125 {
126  mtxMem.lock();
127  mem->Lock();
128 
129  verify(mem->GetEngine() == this);
130 
131  size_t size;
132  void **ptr;
133 
134  ptr = mem->GetPtrs();
135  if(ptr[loc] != NULL)
136  {
137  mtxMem.unlock();
138  mem->Unlock();
139  return;
140  }
141 
142  size = mem->GetSize();
143  verify(size > 0);
144 
145 
146 #ifdef FRACTAL_USE_CUDA
147  if(loc == hostLoc)
148  {
149  verify(cudaMallocHost(ptr + loc, size) == cudaSuccess);
150  }
151  else
152  {
153  size = (size + CUDA_CHUNK_SIZE - 1) / CUDA_CHUNK_SIZE * CUDA_CHUNK_SIZE;
154  verify(cudaMalloc(ptr + loc, size) == cudaSuccess);
155  }
156 #else
157  verify((ptr[loc] = malloc(size)) != NULL);
158 #endif /* FRACTAL_USE_CUDA */
159 
160  memAllocCount++;
161 
162  mtxMem.unlock();
163  mem->Unlock();
164 }

209 void Engine::MemPull(Mem *mem, const unsigned long loc, PStream &stream)
210 {
211  mtxMem.lock();
212  mem->Lock();
213 
214  verify(mem->GetEngine() == this);
215 
216  unsigned long recent;
217 
218  if(mem->IsValid(loc) == true)
219  {
220  mtxMem.unlock();
221  mem->Unlock();
222  return;
223  }
224 
225  recent = mem->GetRecentLoc();
226 
227  //verify(mem->IsValid(recent) == true);
228 
229  MemAlloc(mem, loc);
230  if(mem->IsValid(recent) == true)
231  MemCopy(mem, 0, recent, mem, 0, loc, mem->GetSize(), stream);
232 
233  mem->Validate(loc);
234 
235  mtxMem.unlock();
236  mem->Unlock();
237 }

336 void Engine::MatMult(Matrix<FLOAT> &A, const bool transA, Matrix<FLOAT> &B, const bool transB, Matrix<FLOAT> &C, const FLOAT alpha, const FLOAT beta, PStream &stream)
337 {
338  verify(A.GetEngine() == this);
339  verify(B.GetEngine() == this);
340  verify(C.GetEngine() == this);
341 
342  verify((transA == false ? A.GetNumCols() : A.GetNumRows()) == (transB == false ? B.GetNumRows() : B.GetNumCols()));
343  verify(C.GetNumRows() == (transA == false ? A.GetNumRows() : A.GetNumCols()));
344  verify(C.GetNumCols() == (transB == false ? B.GetNumCols() : B.GetNumRows()));
345 
346  Mem *memA, *memB, *memC;
347  FLOAT *ptrA, *ptrB, *ptrC;
348  unsigned long loc;
349 
350  verify((memA = A.GetMem()) != NULL);
351  verify((memB = B.GetMem()) != NULL);
352  verify((memC = C.GetMem()) != NULL);
353 
354  loc = stream.loc;
355 
356  memA->Pull(loc, stream);
357  memB->Pull(loc, stream);
358 
359  if(beta == (FLOAT) 0 || C.GetNumRows() * C.GetNumCols() * sizeof(FLOAT) < memC->GetSize()) /* Submatrix */
360  memC->Pull(loc, stream);
361  else
362  MemAlloc(memC, loc);
363 
364  ptrA = (FLOAT *)memA->GetPtr(loc) + A.GetOffset();
365  ptrB = (FLOAT *)memB->GetPtr(loc) + B.GetOffset();
366  ptrC = (FLOAT *)memC->GetPtr(loc) + C.GetOffset();
367 
368 #ifdef FRACTAL_USE_CUDA
369  verify(cublasSetStream(cublasHandle, stream.cudaStream) == CUBLAS_STATUS_SUCCESS);
370 
371  if(C.GetNumCols() == 1) /* If C is a vector */
372  {
373  verify(GEMV(cublasHandle,
374  transA == true ? CUBLAS_OP_T : CUBLAS_OP_N,
375  A.GetNumRows(),
376  A.GetNumCols(),
377  &alpha,
378  ptrA,
379  A.GetNumRows(),
380  ptrB,
381  1,
382  &beta,
383  ptrC,
384  1)
385  == CUBLAS_STATUS_SUCCESS);
386  }
387  else
388  {
389  verify(GEMM(cublasHandle,
390  transA == true ? CUBLAS_OP_T : CUBLAS_OP_N,
391  transB == true ? CUBLAS_OP_T : CUBLAS_OP_N,
392  C.GetNumRows(),
393  C.GetNumCols(),
394  transA == true ? A.GetNumRows() : A.GetNumCols(),
395  &alpha,
396  ptrA,
397  A.GetNumRows(),
398  ptrB,
399  B.GetNumRows(),
400  &beta,
401  ptrC,
402  C.GetNumRows())
403  == CUBLAS_STATUS_SUCCESS);
404  }
405 #else
406  verify(false); /* CPU computation is not supported */
407 #endif /* FRACTAL_USE_CUDA */
408 
409  memC->Push(loc);
410 }
