195 void Connection::Forward(const unsigned long batchFrom, const unsigned long batchTo, const unsigned long nStream)
196 {
197  unsigned long actFrom, actTo, delay;
198 
199  verify(batchFrom >= 0 && batchTo < batchSize && batchFrom <= batchTo);
200  verify(engine != NULL);
201 
202 #ifdef FRACTAL_VERBOSE
203  printf("Connection::Forward: %s -> %s %s (%ld, %ld)\n",
204  srcLayer->GetName().c_str(), dstLayer->GetName().c_str(),
205  IsDelayed() == true ? "(DELAYED)" : "",
206  batchFrom, batchTo);
207 #endif /* FRACTAL_VERBOSE */
208 
209  //srcLayer->StreamWaitEvent(*stream);
210 
211  if(IsDelayed() == true)
212  {
213  delay = IsDelayed() == true ? nStream * delayAmount : 0;
214 
215  actFrom = (batchFrom + batchSize - delay) % batchSize;
216  actTo = (batchTo + batchSize - delay) % batchSize;
217 
218  verify(actFrom >= 0 && actTo < batchSize && actFrom <= actTo);
219 
220  Matrix<FLOAT> actSub(srcLayer->act, actFrom, actTo);
221  Matrix<FLOAT> srcActSub(srcAct, batchFrom, batchTo);
222 
223  engine->MatCopy(actSub, srcActSub, *stream);
224  }
225  else
226  {
227  srcAct.Link(srcLayer->act);
228  }
229 
230  if(IsIdentity() == true)
231  {
232  dstAct.Link(srcAct);
233  }
234  else
235  {
236  Matrix<FLOAT> srcActSub(srcAct, batchFrom, batchTo);
237  Matrix<FLOAT> dstActSub(dstAct, batchFrom, batchTo);
238 
239  engine->MatMult(weights, false, srcActSub, false, dstActSub, (FLOAT) 1, (FLOAT) 0, *stream);
240  //if(IsDelayed() == true)
241  // engine->FuncBoundRange(dstActSub, dstActSub, (FLOAT) -10, (FLOAT) 10, *stream);
242  }

316 void Connection::UpdateWeights(const unsigned long batchFrom, const unsigned long batchTo, const unsigned long nFrame,
317  const FLOAT rate, const FLOAT momentum, const bool adadelta, const bool rmsprop)
318 {
319  verify(batchFrom >= 0 && batchTo < batchSize && batchFrom <= batchTo);
320  verify(engine != NULL);
321 
322  if(IsIdentity() == true) return;
323 
324 
325 #ifdef FRACTAL_VERBOSE
326  printf("Connection::UpdateWeights: %s -> %s (%ld, %ld)\n",
327  srcLayer->GetName().c_str(), dstLayer->GetName().c_str(),
328  batchFrom, batchTo);
329 #endif /* FRACTAL_VERBOSE */
330 
331  Matrix<FLOAT> srcActSub(srcAct, batchFrom, batchTo);
332  Matrix<FLOAT> dstErrSub(dstErr, batchFrom, batchTo);
333 
334  /* Simplified Nesterov momentum (Yoshua Bengio) */
335  engine->MatAdd(vels, weights, -momentum, *stream);
336 
337  if(adadelta == true)
338  {
339  verify(rmsprop == false);
340 
341  engine->MatMult(dstErrSub, false, srcActSub, true, derivs, (FLOAT) 1, (FLOAT) 0, *stream);
342  engine->Adadelta(derivs, derivs, msDeriv, msDelta, rate, rmsDecayRate, *stream);
343  engine->MatAdd(vels, vels, momentum - (FLOAT) 1, *stream); // vels *= momentum
344  engine->MatAdd(derivs, vels, (FLOAT) 1, *stream); // vels += rate * derivs
345  }
346  else if(rmsprop == true)
347  {
348  //engine->MatMult(dstErrSub, false, srcActSub, true, derivs, (FLOAT) 1 / (FLOAT) nFrame, (FLOAT) 0, *stream);
349  engine->MatMult(dstErrSub, false, srcActSub, true, derivs, (FLOAT) 1, (FLOAT) 0, *stream);
350  engine->Rmsprop(derivs, derivs, msDeriv, rmsDecayRate, *stream);
351  engine->MatAdd(vels, vels, momentum - (FLOAT) 1, *stream); // vels *= momentum
352  engine->MatAdd(derivs, vels, rate, *stream); // vels += rate * derivs
353  }
354  else
355  {
356  /* vels = momentum * vels + rate * derivs */
357  //engine->MatMult(dstErrSub, false, srcActSub, true, vels, rate / (FLOAT) nFrame, momentum, *stream);
358  engine->MatMult(dstErrSub, false, srcActSub, true, vels, rate, momentum, *stream);
359  }
360  engine->MatAdd(vels, weights, (FLOAT) 1 + momentum, *stream);
361 
362  weightsTransValid = false;
363 }

481 
482  pos2 = buf.find_last_not_of(" \n\r\t", pos3 - 1);
483  pos4 = buf.find_first_not_of(" \n\r\t", pos3 + 1);
484 
485  bufLHS = buf.substr(pos1, pos2 - pos1 + 1);
486  bufRHS = buf.substr(pos4, pos5 - pos4 + 1);
487 
488  //std::cout << bufLHS << "=" << bufRHS << std::endl;
489 
490  if(bufLHS == "rmsDecayRate")
491  {
492  std::istringstream(bufRHS) >> rmsDecayRate;
