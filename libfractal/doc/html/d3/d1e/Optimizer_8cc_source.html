215 void Optimizer::PrepareData(Stream &stream, std::vector<Matrix<FLOAT>> &input, std::vector<Matrix<FLOAT>> &target,
216  const std::vector<unsigned long> &inputChannel, const std::vector<unsigned long> &outputChannel,
217  const unsigned long nFramePerStream)

225  nOutput = outputChannel.size();
226 
227  for(streamIdx = 0; streamIdx < nStream; streamIdx++)
228  {
229  for(i = 0; i < nFramePerStream; i++)
230  {
231  for(j = 0; j < nInput; j++)

245  stream.Next(streamIdx);

247  }
248

251  input[i].HostPush();
252  }
