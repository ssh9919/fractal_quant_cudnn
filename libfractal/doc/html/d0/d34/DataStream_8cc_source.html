169 void DataStream::Next(const unsigned long streamIdx)
170 {
171  unsigned long channelIdx, curSeqIdx, curFrameIdx, curBufIdx, curDim;
172  FLOAT *curBuf;
173 
174  verify(dataSet != NULL);
175 
176  curSeqIdx = seqIdx[streamIdx];
177  curFrameIdx = frameIdx[streamIdx];
178 
179  /* Copy and store delayed frames */
180  for(channelIdx = 0; channelIdx < nChannel; channelIdx++)
181  {
182  if(delay[channelIdx] == 0) continue;
183 
184  curBufIdx = bufIdx[streamIdx][channelIdx];
185  curDim = dim[channelIdx];
186  curBuf = buf[streamIdx][channelIdx].data() + curBufIdx * curDim;
187 
188  dataSet->GetFrameData(curSeqIdx, channelIdx, curFrameIdx, curBuf);
189 
190  bufIdx[streamIdx][channelIdx] = (curBufIdx + 1) % delay[channelIdx];
191  }
192 
193 
194  /* Increase the indices */
195  frameIdx[streamIdx]++;
196 
197  if(frameIdx[streamIdx] == dataSet->GetNumFrame(seqIdx[streamIdx]))
198  {
199  NewSeq(streamIdx);
200  }
